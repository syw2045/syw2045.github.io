int result[MAX][MAX];

	visited[x][y] = true;

	for (int i = 0; i < 4; i++)
	{
		int nx = x + dx[i];
		int ny = y + dy[i];

	{
		for (int j = 0; j < col; j++)
		{

		}

	{
		for (int j = 0; j < col; j++)
