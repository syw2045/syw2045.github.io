#include <iostream>
#include <queue>

#define MAX 1001

using namespace std;

int matrix[MAX][MAX];
bool visited[MAX][MAX];

int row, col;
int result;

queue<pair<int, int>> q;

int dx[4] = { -1, 0, 0, 1 };
int dy[4] = { 0, 1, -1, 0 };

void BFS() // BFS
{
	while (!q.empty())
	{
		int x = q.front().first;
		int y = q.front().second;
		
		q.pop();
		visited[x][y] = true;
		
		for (int i = 0; i < 4; i++)
		{
			int nx = x + dx[i];
			int ny = y + dy[i];

			if (nx < 0 || ny < 0 || nx >= row || ny >= col) continue;
			if (visited[nx][ny] == false && matrix[nx][ny] == 0)
			{
				q.push(make_pair(nx, ny));
				visited[nx][ny] = true;
				matrix[nx][ny] = matrix[x][y] + 1;
			}
		}
	}
}

int main()
{
	cin >> col >> row; // ì—´ , í–‰
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			cin >> matrix[i][j]; // ë°ì´í„° ì…ë ¥

		}
	}

	BFS();

	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			if (matrix[i][j] == 0) // ë§Œì•½ í† ë§ˆí† ê°€ ìµì§€ ì•Šì•˜ë‹¤ë©´
			{
				cout << "-1\n";
				return 0;

			{
				result = matrix[i][j]; // ìµœëŒ€ ê°’ì„ ì°¾ëŠ”ë‹¤
			}
		}
	}

	cout << result - 1 << "\n"; // ì‹œì‘ì„ "1" ë¶€í„° í–ˆê¸° ë•Œë¬¸ì— ë§ˆì§€ë§‰ì— 1ì„ ë¹¼ì¤€ë‹¤.
	return 0;
}
