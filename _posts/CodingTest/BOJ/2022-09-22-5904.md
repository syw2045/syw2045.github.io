int moo = 0;

{
		if (len < moo) // first
		{
			total = moo;
			middle--;
			moo = (total - middle) / 2;
			moo_func(len);
		}
		else if (len > moo + middle) // third
		{
			len -= moo + middle;
			total = moo;
			middle--;
			moo = (total - middle) / 2;
			moo_func(len);
		}
		else // second
		{
			len -= moo;
			total = middle;
			if (len == 1) return 'm';
			else return 'o';
		}
		if (len <= 3)
		{
			if (len == 1) return 'm';
			else return 'o';
		}
}


void cal_len(int N)
{
	while (1)
	{
		total = moo * 2 + middle;
		if (N < total) break;
		moo = total;
		middle++;
	}
}

int main()

	int N = 0;
	cin >> N;
	cal_len(N);

void Cal_MOO(int N, int cnt, int len)
{
