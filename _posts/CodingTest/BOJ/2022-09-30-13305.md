	cin >> N;

	long long total = 0;
	long long value = 0;

	for (long long i = 0; i < N-1; i++)
	{
		cin >> value;
		Road.push_back(value);
	}
	Road.push_back(0);

	for (long long i = 0; i < N; i++)
	{
		cin >> value;
		Oil.push_back(value);
	}

	for (long long i = 0; i < N-1; i++)
	{
		if (Oil[i] < Oil[i + 1])
		{
			total += Oil[i] * (Road[i] + Road[i + 1]);
			i++;
		}
		else
		{
			total += Oil[i] * Road[i];
		}
	}

	cout << total << "\n";

	return 0;
}
